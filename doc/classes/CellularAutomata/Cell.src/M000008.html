                def surround
                        unless @surround
                                #grab unique reactants
                                cells = @reactants
                                #compute amounts
                                cells = cells.map { |c| c.grid.to_s + ":" + c.state.identifier }
                                states = cells.uniq
                                amounts = states.map {|s| (cells.select { |c| c==s }).size }
                                @surround = states.zip amounts
                                @surround.map! { |i| i.join("=") }
                                @surround.map! { |i| Input.new(i) }
                        end
                        @surround
                end
