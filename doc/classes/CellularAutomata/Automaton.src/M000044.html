                def NextState(cell, input)
                        trans = @transitions.select { |t| t.match(cell,input) }
                        if trans.size > 1
                        #    raise "Multiple transitions matched!"
                        end
                        return trans[-1] unless trans.empty?
                        #will only occur with incomplete transition tables
                        return false
                end
